# TTL constants (adjust as needed)

STATE_TTL = 3600      # 1 hour for ingestion state
LOCK_TTL = 300        # 5 minutes for distributed lock
LOCK_RETRY_INTERVAL = 0.5   # seconds between lock wait polls
LOCK_WAIT_TIMEOUT = 120     # max seconds to wait for another worker’s lock

async def ensure_ingested(self, chip_id: int, version: str) -> None:
“””
Ensures chip version data is ingested with distributed locking.

```
Fast-path: returns immediately if already READY.
Lock-path: acquires distributed lock, ingests if needed.
Wait-path: if lock held by another worker, waits for completion.

Args:
    chip_id: The chip identifier
    version: The version string (e.g., "5.5")
"""
version_key = self._version_key(chip_id, version)
lock_key = self._lock_key(chip_id, version)

logger.info(f"Ensuring ingestion for chip={chip_id}, version={version}")

# ─── Fast path: already ingested ─────────────────────────────────────────
state = await self.redis.get(version_key)

if state == IngestionState.READY:
    logger.debug(f"Fast path: data already READY for {chip_id}:{version}")
    return

# ─── State is None: check if underlying data exists ──────────────────────
if state is None:
    data_exists = await self._check_data_exists(chip_id, version)
    if data_exists:
        # Data exists but state key is missing (e.g. TTL expired or redis flush)
        # Acquire lock before marking READY to avoid races
        lock_acquired = await self.redis.set(
            lock_key, "1", nx=True, ex=LOCK_TTL
        )
        if lock_acquired:
            try:
                logger.info(
                    f"Data exists but state missing, marking READY for "
                    f"{chip_id}:{version}"
                )
                await self.redis.set(version_key, IngestionState.READY, ex=STATE_TTL)
            finally:
                await self.redis.delete(lock_key)
        else:
            # Another worker is handling it — just wait
            await self._wait_for_ingestion(version_key, chip_id, version)
        return

# ─── FAILED state: log warning, fall through to re-ingest ────────────────
if state == IngestionState.FAILED:
    logger.warning(
        f"Previous ingestion FAILED for {chip_id}:{version}, retrying..."
    )

# ─── Acquire distributed lock to begin ingestion ─────────────────────────
lock_acquired = await self.redis.set(
    lock_key, "1", nx=True, ex=LOCK_TTL
)

if lock_acquired:
    logger.info(f"Lock acquired for {chip_id}:{version}, starting ingestion")
    try:
        await self._perform_ingestion(chip_id, version, version_key, lock_key)
    except Exception as e:
        # Mark as FAILED so other workers/retries know
        await self.redis.set(version_key, IngestionState.FAILED, ex=STATE_TTL)
        logger.error(
            f"Ingestion failed for {chip_id}:{version}: {e}", exc_info=True
        )
        raise
    finally:
        await self.redis.delete(lock_key)
else:
    # Another worker holds the lock — wait for it to finish
    logger.info(
        f"Lock held by another worker, waiting for {chip_id}:{version}"
    )
    await self._wait_for_ingestion(version_key, chip_id, version)

async def _wait_for_ingestion(
self,
version_key: str,
chip_id: int,
version: str,
) -> None:
“””
Polls Redis until ingestion is READY or FAILED, or timeout is reached.

```
Args:
    version_key: Redis key to poll for state
    chip_id: Used for logging
    version: Used for logging

Raises:
    TimeoutError: If ingestion doesn't complete within LOCK_WAIT_TIMEOUT
    RuntimeError: If the ingestion performed by another worker FAILED
"""
elapsed = 0.0

while elapsed < LOCK_WAIT_TIMEOUT:
    await asyncio.sleep(LOCK_RETRY_INTERVAL)
    elapsed += LOCK_RETRY_INTERVAL

    state = await self.redis.get(version_key)

    if state == IngestionState.READY:
        logger.info(
            f"Ingestion completed by another worker for {chip_id}:{version} "
            f"(waited {elapsed:.1f}s)"
        )
        return

    if state == IngestionState.FAILED:
        raise RuntimeError(
            f"Ingestion failed (performed by another worker) for "
            f"{chip_id}:{version}"
        )

raise TimeoutError(
    f"Timed out waiting for ingestion of {chip_id}:{version} "
    f"after {LOCK_WAIT_TIMEOUT}s"
)

async def _handle_qgenie_function_call(
    self,
    content: str,
    chat_history: "ChatHistory",
    settings: "PromptExecutionSettings",
    kernel: Any,
    messages: List[Dict],
    max_iterations: int = 5,
) -> ChatMessageContent:
    """
    Handle QGenie's custom function call format with multiple iterations.

    Per-iteration flow:
      1.  Detect tool call token in current_content
      2.  Parse plugin_name + function_name + arguments         (BUG 1 FIX)
      3.  Verify function exists in kernel with BOTH names      (BUG 2 FIX)
      4.  Execute via kernel.invoke(plugin_name, function_name)
      5.  Append result as role="tool" NOT role="user"          (BUG 3 FIX)
      6.  Call QGenie to summarise
      7.  If clean response → return immediately                (BUG 4 FIX)
      8.  Otherwise loop

    BUG 7 FIX: "import re" has been moved to top of module (was in this loop).
    """
    iteration       = 0
    current_content = content

    while iteration < max_iterations:
        iteration += 1
        logger.info("Function call iteration %d/%d", iteration, max_iterations)

        # ── Step 1: Check for function call token ─────────────────────────
        if not _is_function_call_response(current_content):
            cleaned = _clean_response(current_content)
            logger.info(
                "No function call detected — returning final response (%d chars)",
                len(cleaned),
            )

            # BUG 5 FIX: only fallback if truly empty, not just short
            if _is_empty_or_metadata_only(current_content):
                logger.warning("Empty/metadata-only response → summary fallback")
                return self._create_summary_from_messages(messages)

            # BUG 4 FIX: explicit early return (original continued the loop)
            return ChatMessageContent(
                role=AuthorRole.ASSISTANT,
                content=cleaned,
            )

        try:
            # ── Step 2: Parse function call ───────────────────────────────

def _sanitise_args(function_name: str, args: Dict[str, Any]) -> Dict[str, Any]:
“””
Clean args before passing to kernel.invoke.

```
ROOT CAUSE of the screenshot error:
  QGenie parsed 'version': None from JSON args.
  SK kernel_function_from_method._parse_parameter() tries to call
  Optional[str](None) to coerce the value.
  typing.Union cannot be instantiated → TypeError: Cannot instantiate typing.Union

Fix — two rules:
  1. Known params with safe defaults: version=None → "latest"
  2. Truly optional params: drop if None so SK uses function signature default
"""
# Replace None with these safe defaults
NONE_DEFAULTS: Dict[str, Any] = {
    "version": "latest",
}

# Drop these if None — SK uses the function's own default value
OPTIONAL_DROP_IF_NONE = {"mpu_name", "region", "address", "region_index"}

sanitised: Dict[str, Any] = {}
for key, value in args.items():
    if value is None:
        if key in NONE_DEFAULTS:
            sanitised[key] = NONE_DEFAULTS[key]
            logger.debug("sanitise_args: %s=None → '%s'", key, NONE_DEFAULTS[key])
        elif key in OPTIONAL_DROP_IF_NONE:
            logger.debug("sanitise_args: dropping %s=None (optional)", key)
            # intentionally omitted from sanitised dict
        else:
            logger.warning(
                "sanitise_args: %s=None for function '%s' — "
                "may cause SK TypeError. Add to NONE_DEFAULTS if needed.",
                key, function_name,
            )
            sanitised[key] = value
    else:
        sanitised[key] = value

logger.info("sanitise_args: %s → %s", args, sanitised)
return sanitised

async def _execute_function(
    self,
    kernel: Any,
    plugin_name: str,
    function_name: str,
    function_args: Dict[str, Any],
) -> str:
    """
    Execute a kernel function safely.

    FIX for screenshot error:
      'version': None in function_args caused:
        TypeError: Cannot instantiate typing.Union
      in SK kernel_function_from_method._parse_parameter()

    Two-layer defence:
      1. _sanitise_args(): None values replaced with safe defaults or dropped.
      2. KernelArguments wrapper: gives SK pre-typed values, avoids raw
         **kwargs unpacking that hits the Union instantiation code path.
    """
    # Layer 1: sanitise None values
    sanitised = _sanitise_args(function_name, function_args)

    # Layer 2: wrap in KernelArguments so SK skips type coercion
    from semantic_kernel.functions.kernel_arguments import KernelArguments
    kernel_args = KernelArguments(**sanitised)

    result = await kernel.invoke(
        plugin_name=plugin_name,
        function_name=function_name,
        arguments=kernel_args,
    )
    return str(result) if result is not None else ""

Root cause of this bug:
QGenie requires FULL plugin-prefixed function calls:
✓  PolicySearch.get_mpu_summary(project, mpu_name, version)
✗  get_mpu_summary(project, mpu_name, version)

The system prompt was listing bare function names.
QGenie’s parser splits on “.” to find plugin + function.
Without the prefix, it can’t resolve the call → falls back
to the default plugin (PolicySearch) for EVERYTHING.

This is why even correct function choices like get_mpu_summary
ended up being called as PolicySearch.get_mpu_summary accidentally,
and wrong ones like count_resource_groups silently failed resolution.

Fix:

1. All function names in system prompt → PluginName.function_name
1. All decision tree rules → PluginName.function_name
1. All few-shot examples   → PluginName.function_name
1. Recommended function nudge in context → PluginName.function_name
1. QGenieResponseParser updated to handle “PluginName.function_name”

def _make_tool_call_message(self, tool_call: dict) -> ChatMessageContent:
    """
    Convert a parsed QGenie tool call into SK FunctionCallContent.

    QGenie response format (after our parser extracts it):
      function = "PolicySearch.get_mpu_summary"   ← already has prefix
      OR
      function = "get_mpu_summary"                ← bare name (fallback)

    SK FunctionCallContent name format: "PluginName-function_name"
    """
    raw_name  = tool_call["function"]    # e.g. "PolicySearch.get_mpu_summary"
    arguments = tool_call.get("arguments", {})

    # ── Split plugin.function correctly ───────────────────────────────────
    if "." in raw_name:
        # QGenie returned the full prefixed name — split it
        plugin_part, function_part = raw_name.split(".", 1)
    else:
        # Bare name — attach the known plugin name
        plugin_part   = PLUGIN_NAME
        function_part = raw_name

    # SK uses "PluginName-function_name" as the FunctionCallContent name
    sk_function_name = f"{plugin_part}-{function_part}"

    logger.info(
        "Tool call: raw='%s' → SK name='%s'  args=%s",
        raw_name, sk_function_name, arguments,
    )

    function_call_content = FunctionCallContent(
        id=f"qgenie_{function_part}_{id(arguments)}",
        name=sk_function_name,
        arguments=json.dumps(arguments),
    )

    return ChatMessageContent(
        role=AuthorRole.ASSISTANT,
        items=[function_call_content],
        name=self.ai_model_id,
    )

TOOL_CALL_PATTERN = re.compile(
    r"<\|start\|>.*?<\|channel\|>.*?"
    r"to=functions\.([\w.]+)"       # FIXED: [\w.]+ captures "PolicySearch.get_mpu_summary"
    r".*?<\|message\|>"
    r"(\{.*?\})"
    r"<\|call\|>",
    re.DOTALL,
)

TOKEN_STRIP_PATTERN = re.compile(r"<\|[^|]+\|>")

@classmethod
def parse(cls, raw_response: str):
    if not raw_response:
        return None, None

    match = cls.TOOL_CALL_PATTERN.search(raw_response)
    if match:
        raw_function_name = match.group(1).strip()
        # e.g. "PolicySearch.get_mpu_summary" or "get_mpu_summary"
        args_json = match.group(2).strip()

        # Normalise: ensure plugin prefix is present
        if "." not in raw_function_name:
            full_function_name = f"{PLUGIN_NAME}.{raw_function_name}"
        else:
            full_function_name = raw_function_name

        try:
            arguments = json.loads(args_json)
        except json.JSONDecodeError:
            logger.warning("Failed to parse tool call args: %s", args_json)
            arguments = {}

        logger.info(
            "QGenie tool call: raw='%s' → resolved='%s'  args=%s",
            raw_function_name, full_function_name, arguments,
        )
        return None, {"function": full_function_name, "arguments": arguments}

    clean_text = cls.TOKEN_STRIP_PATTERN.sub("", raw_response).strip()
    return clean_text or None, None

“””
Decide if query needs auto function calling (multi-step SK tool execution).

```

cause false positives and trigger auto function calling on simple queries.
"""
query_lower = query.lower().strip()

# ── 1. MULTI-STEP indicators ──────────────────────────────────────────────
# User explicitly wants sequential actions
multi_step_indicators = [
    "and then",
    "after that",
    "followed by",
    "first.*then",      # regex-style intent: "first do X then do Y"
]

# ── 2. COMPARISON indicators ──────────────────────────────────────────────
# User wants to compare two or more things — always needs multiple tool calls
comparison_indicators = [
    "compare",
    "difference between",
    "versus",
    " vs ",
    "which has more",
    "which has less",
    "which is higher",
    "which is lower",
    "better than",
    "worse than",
]

# ── 3. AGGREGATION indicators ─────────────────────────────────────────────
# User wants rolled-up data across multiple records
aggregation_indicators = [
    "total count",
    "sum of",
    "average of",
    "across all projects",
    "across all versions",
    "for every project",
    "for each project",
]

# ── 4. FILTERING + RANKING indicators ────────────────────────────────────
# Specific filtering that needs structured tool calls
filter_rank_indicators = [
    "filter by",
    "where project",
    "that have more than",
    "that have less than",
    "exclude",
    "top 5",
    "top 10",
    "bottom 5",
    "highest mpu",
    "lowest mpu",
    "most regions",
    "fewest regions",
]

# ── 5. DETAILS / DRILL-DOWN indicators ───────────────────────────────────
# Queries explicitly asking for deep detail on specific entities
detail_indicators = [
    "show me details for",
    "get details for",
    "details of",
    "information about",
    "breakdown of",
    "full details",
    "drill down",
]

# ── 6. ANALYSIS indicators ────────────────────────────────────────────────
analysis_indicators = [
    "analyze",
    "analyse",
    "give me an overview of",
    "statistics for",
    "summarize all",          # NOT just "summary" — too broad
    "overview of all",
]

# ── Combine all structured indicators ────────────────────────────────────
all_indicators = (
    multi_step_indicators
    + comparison_indicators
    + aggregation_indicators
    + filter_rank_indicators
    + detail_indicators
    + analysis_indicators
)

# ── Check for indicator match ─────────────────────────────────────────────
matched_indicator = None
for indicator in all_indicators:
    if indicator in query_lower:
        matched_indicator = indicator
        break

is_complex = matched_indicator is not None

# ── Check for multiple entities ───────────────────────────────────────────
# FIX: was > 2 (needs 3+), should be >= 2 (2 entities = comparison needed)
has_multiple_entities = len(facts.entity) >= 2

# ── Decision with detailed logging ───────────────────────────────────────
if is_complex:
    logger.info(
        "Auto function calling: COMPLEX query detected  "
        "indicator='%s'  query='%s'",
        matched_indicator, query[:80],
    )
    return True

if has_multiple_entities:
    logger.info(
        "Auto function calling: MULTIPLE ENTITIES detected  "
        "entity_count=%d  entities=%s  query='%s'",
        len(facts.entity), facts.entity, query[:80],
    )
    return True

logger.debug(
    "Auto function calling: NOT needed  "
    "is_complex=%s  entity_count=%d  query='%s'",
    is_complex, len(facts.entity), query[:80],
)
return False

# normalizer.py

import re
from rapidfuzz import process
from typing import List


class QueryNormalizer:

    def __init__(self, valid_projects: List[str], valid_mpus: List[str]):
        self.valid_projects = valid_projects
        self.valid_mpus = valid_mpus

    # -----------------------------
    # Main Entry
    # -----------------------------
    def normalize(self, query: str) -> str:

        query = query.strip()
        query = self._normalize_case(query)
        query = self._normalize_hex(query)
        query = self._normalize_keywords(query)
        query = self._normalize_projects(query)
        query = self._normalize_mpus(query)

        return query

    # -----------------------------
    # Lowercase but preserve hex
    # -----------------------------
    def _normalize_case(self, query: str) -> str:
        return query.lower()

    # -----------------------------
    # Normalize hex
    # -----------------------------
    def _normalize_hex(self, query: str) -> str:

        hex_pattern = r"\b([0-9a-f]{6,16})\b"

        def replace_hex(match):
            value = match.group(1)
            return "0x" + value.upper()

        return re.sub(hex_pattern, replace_hex, query)

    # -----------------------------
    # Normalize keywords
    # -----------------------------
    def _normalize_keywords(self, query: str) -> str:

        replacements = {
            "fetch": "show",
            "give": "show",
            "display": "show",
            "retrieve": "show"
        }

        for k, v in replacements.items():
            query = query.replace(k, v)

        return query

    # -----------------------------
    # Fuzzy project matching
    # -----------------------------
    def _normalize_projects(self, query: str) -> str:

        for word in query.split():
            match = process.extractOne(word, self.valid_projects, score_cutoff=80)
            if match:
                correct = match[0]
                query = query.replace(word, correct)

        return query

    # -----------------------------
    # Fuzzy MPU matching
    # -----------------------------
    def _normalize_mpus(self, query: str) -> str:

        for word in query.split():
            match = process.extractOne(word, self.valid_mpus, score_cutoff=75)
            if match:
                correct = match[0]
                query = query.replace(word, correct)

        return query

import re
from enum import Enum

class RouteType(str, Enum):
    TAG = "TAG"
    LLM = "LLM"

class QueryRouter:
    # 1. Strict Technical Keywords (including your plural/irregular requirements)
    TAG_PATTERNS = [
        r"polic(?:y|ies)", 
        r"mpus?", 
        r"regions?", 
        r"projects?",
        r"address(?:es)?", 
        r"profiles?", 
        r"counts?", 
        r"lists?", 
        r"versions?"
    ]
    
    # 2. Conceptual/Explainer words that force an LLM route
    LLM_INTENT_PATTERNS = [
        r"\bwhy\b", r"\bhow\b", r"\bexplain\b", r"\bdifference\b", r"\bwhat is\b"
    ]

    # Compiled with custom boundaries to handle underscores like CNOC_SS_MPU
    TAG_REGEX = re.compile(
        rf"(?<![a-zA-Z0-9])(?:{'|'.join(TAG_PATTERNS)})(?![a-zA-Z0-9])", 
        re.IGNORECASE
    )
    
    LLM_INTENT_REGEX = re.compile(
        rf"(?:{'|'.join(LLM_INTENT_PATTERNS)})", 
        re.IGNORECASE
    )

    ADDRESS_PATTERN = re.compile(r"\b0x[0-9a-f]+\b", re.IGNORECASE)

    @classmethod
    def route(cls, query: str) -> RouteType:
        # Step 1: Detect if the query has TAG-specific markers (Address or Keywords)
        has_tag_marker = cls.ADDRESS_PATTERN.search(query) or cls.TAG_REGEX.search(query)
        
        # Step 2: Detect if the user is asking "Why/How/Explain"
        is_conceptual = cls.LLM_INTENT_REGEX.search(query)

        # Logic: ONLY route to TAG if markers exist AND it's not a conceptual question.
        if has_tag_marker and not is_conceptual:
            return RouteType.TAG

        # Step 3: Default Fallback for everything else (Hello, Time, Why MPU, etc.)
        return RouteType.LLM

async def handle_message(message: cl.Message):

    clarify_state = cl.user_session.get("clarify_state")

    # -------------------------------------------------
    # STEP 1: If user is answering clarification
    # -------------------------------------------------
    if clarify_state:
        # Merge missing field
        clarify_state["entities"][clarify_state["awaiting"]] = message.content
        cl.user_session.set("clarify_state", None)

        response = await orchestrator.process(
            clarify_state["query"],
            clarify_state["entities"]
        )
    else:
        # Normal flow
        entities = await orchestrator.extract_entities(message.content)

        response = await orchestrator.process(
            message.content,
            entities
        )

    # -------------------------------------------------
    # STEP 2: Route Based on Typed Response
    # -------------------------------------------------

    if response.type == "CLARIFY":

        missing_field = response.missing[0]

        cl.user_session.set(
            "clarify_state",
            {
                "query": message.content,
                "entities": entities,
                "awaiting": missing_field,
            }
        )

        await cl.Message(
            content=f"Please provide **{missing_field}**."
        ).send()
        return

    if response.type == "TAG":

        await cl.Message(
            content=response.data
        ).send()
        return

    if response.type == "LLM":

        # LLM memory only here
        memory.append(session_id, "user", message.content)

        history = memory.get(session_id)

        answer = await orchestrator.llm_response(history)

        memory.append(session_id, "assistant", answer)
        memory.trim(session_id)

        await cl.Message(content=answer).send()
        return

def _version_key(self, chip, version):
    return f"ipcat:version:{chip}:{version}"

def _lock_key(self, chip, version):
    return f"ipcat:lock:{chip}:{version}"

ipcat:version:{chip}:{version} -> READY | IN_PROGRESS | FAILED
ipcat:lock:{chip}:{version}    -> distributed lock
ipcat:version_ts:{chip}:{version} -> ingestion timestamp


import asyncio
import time

LOCK_TTL = 300
WAIT_TIMEOUT = 60
POLL_INTERVAL = 1


async def _ensure_ingested(self, chip: str, version: str):

    version_key = self._version_key(chip, version)
    lock_key = self._lock_key(chip, version)

    state = self.redis.get(version_key)

    # ✅ Fast path
    if state == "READY":
        return

    # 🔒 Try acquiring distributed lock
    lock_acquired = self.redis.set(
        lock_key,
        "1",
        nx=True,
        ex=LOCK_TTL,
    )

    if lock_acquired:
        # We are the ingestion owner
        try:
            self.redis.set(version_key, "IN_PROGRESS")

            await self._ingest_chip(chip, version)

            self.redis.set(version_key, "READY")
            self.redis.set(f"{version_key}:ts", int(time.time()))

        except Exception as e:
            self.redis.set(version_key, "FAILED")
            raise

        finally:
            self.redis.delete(lock_key)

    else:
        # Another worker is ingesting
        await self._wait_for_ingestion(version_key)


async def _wait_for_ingestion(self, version_key):

    start = time.time()

    while time.time() - start < WAIT_TIMEOUT:

        state = self.redis.get(version_key)

        if state == "READY":
            return

        if state == "FAILED":
            raise RuntimeError("Ingestion failed")

        await asyncio.sleep(POLL_INTERVAL)

    raise TimeoutError("Timed out waiting for ingestion")


async def search_by_address(self, chip, version, mpu, address):

    # 1️⃣ Ensure ingestion
    await self._ensure_ingested(chip, version)

    # 2️⃣ Load interval tree
    tree = self._load_interval_tree(chip, version, mpu)

    if not tree:
        return []

    # 3️⃣ Query
    matches = tree[address]

    return [interval.data for interval in matches]

# -----------------------------
# QueryFacts (Aligned Contract)
# -----------------------------

@dataclass
class QueryFacts:
    chip: Optional[str] = None
    version: Optional[str] = None
    mpu: Optional[str] = None
    region_index: Optional[int] = None
    address_start: Optional[int] = None
    address_end: Optional[int] = None
    profile: Optional[str] = None
    operation: str = "LOOKUP"   # LOOKUP | LIST | COUNT
    entity: str = "POLICY"      # POLICY | PROJECT

class UnifiedPolicyEngine:

    def __init__(self, redis_client: redis.Redis):
        self.redis = redis_client

    # =============================
    # Public Entry
    # =============================

    def search(self, facts: QueryFacts) -> Dict[str, Any]:

        if facts.entity == "PROJECT":
            return self._handle_project_query(facts)

        # POLICY level queries
        return self._handle_policy_query(facts)

    # =============================
    # Project Queries
    # =============================

    def _handle_project_query(self, facts: QueryFacts):

        keys = self.redis.keys("ipcat:chip:*")

        projects = [k.decode().split(":")[-1] for k in keys]

        if facts.operation == "COUNT":
            return {"count": len(projects)}

        return {"projects": projects}

    # =============================
    # Policy Queries
    # =============================

    def _handle_policy_query(self, facts: QueryFacts):

        if not facts.chip:
            raise ValueError("Chip required for policy search")

        version = self._resolve_version(facts)

        if facts.region_index is not None:
            return self._search_by_region(facts, version)

        if facts.address_start is not None:
            return self._search_by_address(facts, version)

        if facts.operation == "LIST":
            return self._list_regions(facts, version)

        if facts.operation == "COUNT":
            return self._count_regions(facts, version)

        raise ValueError("Unsupported policy query")

    # =============================
    # Region Lookup
    # =============================

    def _search_by_region(self, facts: QueryFacts, version: str):

        pattern = f"ipcat:region:{facts.chip}:{version}:{facts.mpu}:{facts.region_index}:*"

        keys = self.redis.keys(pattern)

        results = [json.loads(self.redis.get(k)) for k in keys]

        return {"rows": results}

    # =============================
    # Address Interval Search
    # =============================

    def _search_by_address(self, facts: QueryFacts, version: str):

        pattern = f"ipcat:region:{facts.chip}:{version}:{facts.mpu}:*"

        keys = self.redis.keys(pattern)

        matched = []

        for key in keys:
            data = json.loads(self.redis.get(key))

            start = int(data["addr_start"], 16)
            end = int(data["addr_end"], 16)

            if start <= facts.address_start <= end:
                if facts.profile and data["profile"] != facts.profile:
                    continue
                matched.append(data)

        return {"rows": matched}

    # =============================
    # LIST Regions
    # =============================

    def _list_regions(self, facts: QueryFacts, version: str):

        pattern = f"ipcat:region:{facts.chip}:{version}:{facts.mpu}:*"

        keys = self.redis.keys(pattern)

        rows = [json.loads(self.redis.get(k)) for k in keys]

        return {"rows": rows}

    # =============================
    # COUNT Regions
    # =============================

    def _count_regions(self, facts: QueryFacts, version: str):

        pattern = f"ipcat:region:{facts.chip}:{version}:{facts.mpu}:*"

        keys = self.redis.keys(pattern)

        return {"count": len(keys)}

    # =============================
    # Version Resolver
    # =============================

    def _resolve_version(self, facts: QueryFacts):

        if facts.version:
            return facts.version

        # fallback latest
        latest_key = f"ipcat:latest:{facts.chip}"
        latest = self.redis.get(latest_key)

        if not latest:
            raise ValueError("Latest version not found")

        return latest.decode()




def parse_regions_from_prtn(
    mpu_elem: ET.Element,
    chip: str,
    version: str,
    mpu_name: str,
) -> List[Region]:

    regions = []

    sw_config = mpu_elem.find("SWConfig")
    if sw_config is None:
        return regions

    for prtn in sw_config.findall("PRTn"):
        try:
            index = int(prtn.attrib.get("index", "-1"))
            start = int(prtn.attrib.get("start", "0"), 16)
            end = int(prtn.attrib.get("end", "0"), 16)
            order = int(prtn.attrib.get("order", "0"))

            profile_str = prtn.attrib.get("profile", "")
            profiles = [
                p.strip().upper()
                for p in profile_str.split(",")
                if p.strip()
            ]

            flags_elem = prtn.find("XPU_RGn_FLAGS")
            flags = flags_elem.attrib if flags_elem is not None else {}

            metadata = {
                "security_rationale":
                    (prtn.findtext("SecurityRationale") or "").strip()
            }

            regions.append(
                Region(
                    chip=chip,
                    version=version,
                    mpu=mpu_name,
                    index=index,
                    start=start,
                    end=end,
                    order=order,
                    profiles=profiles,
                    flags=flags,
                    metadata=metadata,
                )
            )

        except Exception as e:
            logger.exception("Failed parsing region: %s", e)

    return regions

from intervaltree import Interval, IntervalTree

def ingest_single_mpu(
    mpu_elem: ET.Element,
    chip: str,
    version: str,
    redis_client,
) -> None:

    mpu_name = (
        mpu_elem.attrib.get("name", "")
        .strip()
        .upper()
    )

    # ---- Store MPU metadata ----
    mpu_key = f"ipcat:mpu:{chip}:{version}:{mpu_name}"

    redis_client.set(
        mpu_key,
        json.dumps({
            "chip": chip,
            "version": version,
            "mpu": mpu_name,
        }),
    )

    # ---- Parse regions ----
    regions = parse_regions_from_prtn(
        mpu_elem,
        chip,
        version,
        mpu_name,
    )

    tree = IntervalTree()

    for region in regions:

        region_key = (
            f"ipcat:region:{chip}:{version}:{mpu_name}:{region.index}"
        )

        redis_client.set(
            region_key,
            json.dumps(region.__dict__),
        )

        # build interval tree
        tree.add(
            Interval(
                region.start,
                region.end + 1,
                region.__dict__,
            )
        )

    # ---- Store serialized interval tree ----
    tree_key = f"ipcat:interval:{chip}:{version}:{mpu_name}"

    serialized = [
        {
            "start": iv.begin,
            "end": iv.end,
            "data": iv.data,
        }
        for iv in tree
    ]

    redis_client.set(tree_key, json.dumps(serialized))

def load_interval_tree(redis_client, chip, version, mpu):
    key = f"ipcat:interval:{chip}:{version}:{mpu}"
    raw = redis_client.get(key)
    if not raw:
        return None

    data = json.loads(raw)

    tree = IntervalTree()

    for item in data:
        tree.add(
            Interval(
                item["start"],
                item["end"],
                item["data"],
            )
        )

    return tree

def search_policy(
    redis_client,
    chip: str,
    version: str,
    mpu: str,
    address: int,
    profile: str | None = None,
    stage: str | None = None,
):
    tree = load_interval_tree(redis_client, chip, version, mpu)
    if not tree:
        return []

    matches = tree[address]

    results = []

    for interval in matches:
        region = interval.data

        # ---- profile filtering ----
        if profile:
            if profile.upper() not in region["profiles"]:
                continue

        # ---- stage filtering (optional future use) ----
        if stage:
            # example: NON_STATIC, TME_FW may imply stage
            if stage.upper() not in region["profiles"]:
                continue

        results.append(region)

    return results

async def fetch_policy_xml(
    self,
    export_id: int,
    token: str,
    timeout: int = 300,
    poll_interval: int = 5,
):
    """
    Poll export job and yield XML pages.
    """

    import time

    start = time.time()

    while True:
        resp = self.request(
            "GET",
            api=f"xpu/policy/export/{export_id}/",
            params={"raw": True, "token": token},
            response="raw",
        )

        if resp.status_code == 202:
            if time.time() - start > timeout:
                raise TimeoutError("Policy export timed out")
            await asyncio.sleep(poll_interval)
            continue

        if resp.status_code != 200:
            raise IPCatalogRequestError(resp.text)

        yield resp.text
        break





import re


async def resolve_chip(self, chip_input: str) -> dict | None:
    """
    Resolve chip based on:
      - base family (kaanapali)
      - optional revision (r2)

    Examples:
      kaanapali        -> base chip
      kaanapali r2     -> revision chip
    """

    if not chip_input:
        return None

    raw = chip_input.lower().strip()

    # ------------------------------------
    # 1. Extract revision (r2, r3, etc.)
    # ------------------------------------
    revision_match = re.search(r"\br\d+\b", raw)
    revision = revision_match.group(0) if revision_match else None

    # Remove revision from base search string
    base_name = re.sub(r"\br\d+\b", "", raw).strip()

    # ------------------------------------
    # 2. Fetch chip list from API
    # ------------------------------------
    resp = self.request(
        "GET",
        f"{self.base_url}/api/1/chip/",
    )

    chips = resp.json()

    if not chips:
        return None

    # ------------------------------------
    # 3. Match by family token
    # ------------------------------------
    family_matches = []

    for chip in chips:
        full_name = chip["name"].lower()

        # Extract family inside parentheses
        match = re.search(r"\((.*?)\)", full_name)
        family = match.group(1).lower() if match else ""

        if base_name == family or base_name in full_name:
            family_matches.append(chip)

    if not family_matches:
        return None

    # ------------------------------------
    # 4. If revision requested → match exact
    # ------------------------------------
    if revision:
        for chip in family_matches:
            if revision in chip["name"].lower():
                return chip

        return None  # revision explicitly requested but not found

    # ------------------------------------
    # 5. No revision → pick base (no rX)
    # ------------------------------------
    for chip in family_matches:
        if not re.search(r"\br\d+\b", chip["name"].lower()):
            return chip

    # Fallback: return first match
    return family_matches[0]

class PolicyOrchestrator:
    def __init__(self, client, redis):
        self.client = client
        self.redis = redis

    # =====================================================
    # PUBLIC ENTRY
    # =====================================================

    async def get_policy_by_mpu(
        self,
        chip_name: str,
        version: str | None,
        mpu_name: str,
    ) -> Dict[str, Any]:
        """
        Main API entry.
        """

        # 1️⃣ Resolve chip via API (NOT Redis)
        chip = await self.client.get_chip_by_name(chip_name)
        if not chip:
            raise ValueError(f"Chip not found: {chip_name}")

        chip_id = chip["id"]

        # 2️⃣ Resolve policy via API
        policy = await self.client.get_policy_for_chip(
            chip_id=chip_id,
            version=version,
        )

        policy_version = policy["version"]

        # 3️⃣ Ensure XML ingestion
        await self._ensure_ingested(chip_id, policy_version)

        # 4️⃣ Fetch MPU from Redis
        key = MPU_KEY.format(
            chip_id=chip_id,
            policy_version=policy_version,
            mpu_name=mpu_name.upper(),
        )

        raw = self.redis.get(key)
        if not raw:
            raise KeyError(f"MPU not found: {mpu_name}")

        return json.loads(raw)

    # =====================================================
    # INGESTION ORCHESTRATION
    # =====================================================

    async def _ensure_ingested(self, chip_id: int, policy_version: str):
        done_key = INGEST_DONE_KEY.format(
            chip_id=chip_id,
            policy_version=policy_version,
        )

        lock_key = INGEST_LOCK_KEY.format(
            chip_id=chip_id,
            policy_version=policy_version,
        )

        # Fast path
        if self.redis.exists(done_key):
            return

        # Acquire lock
        if not self.redis.set(lock_key, 1, nx=True, ex=LOCK_TTL):
            return

        try:
            await self._ingest_chip(chip_id, policy_version)
            self.redis.set(done_key, 1, ex=DONE_TTL)
        finally:
            self.redis.delete(lock_key)

    async def _ingest_chip(self, chip_id: int, policy_version: str):
        """
        Fetch XML → Parse MPUs → Store each MPU
        """

        async for xml_text in self.client.fetch_policy_xml(
            chip_id,
            policy_version,
        ):
            mpus = extract_mpus(xml_text)

            loop = asyncio.get_running_loop()
            tasks = [
                loop.run_in_executor(
                    EXECUTOR,
                    ingest_single_mpu,
                    xml_text,
                    chip_id,
                    policy_version,
                    mpu,
                    self.redis,
                )
                for mpu in mpus
            ]

            for t in asyncio.as_completed(tasks):
                await t

def extract_hw_params(mpu_elem):
    result = {}

    hw = mpu_elem.find("HwConfig")
    if hw is None:
        return result

    design = hw.find("HwDesignParameters")
    if design is not None:
        result["design"] = {
            p.attrib["name"]: p.attrib["value"]
            for p in design.findall("Parameter")
        }

    integration = hw.find("HwIntegrationParameters")
    if integration is not None:
        result["integration"] = {
            p.attrib["name"]: p.attrib["value"]
            for p in integration.findall("Parameter")
        }

    return result

import re

def _normalize(s: str) -> str:
    return re.sub(r"[^a-z0-9]+", "", s.lower())


def _extract_alias(name: str) -> Optional[str]:
    """
    Examples:
      "SM8850 (KaanapaliT)" -> kaanapali
      "SM8850 (KaanapaliS)" -> kaanapali
      "Zealis v3"           -> zealis
    """
    m = re.search(r"\(([^)]+)\)", name)
    if m:
        base = m.group(1)
        base = re.sub(r"(tsmc|samsung|[ts])$", "", base.lower())
        return _normalize(base)

    # fallback: first token
    return _normalize(name.split()[0])


def _extract_family(name: str) -> Optional[str]:
    # For now: family == alias
    return _extract_alias(name)



from packaging.version import Version
import json

def _resolve_chip_id(
    self,
    chip_name: str,
    version: str | None = None,
) -> str | None:
    """
    Resolve chip name / alias / family → chip_id.
    If version is None → pick latest active.
    """

    name = chip_name.lower().strip()

    # ------------------------------------------------
    # 1. Fast path: direct name/alias index
    # ------------------------------------------------
    chip_id = self.redis.get(
        CHIP_NAME_KEY.format(name=name)
    )
    if chip_id:
        return chip_id.decode() if isinstance(chip_id, bytes) else chip_id

    # ------------------------------------------------
    # 2. Fallback: family-based resolution
    # ------------------------------------------------
    raw = self.redis.get(CHIP_LIST_KEY)
    if not raw:
        return None

    chips = json.loads(raw)

    # Match by family OR name token
    candidates = [
        c for c in chips
        if (
            c.get("family", "").lower() == name
            or name in c.get("name", "").lower()
        )
    ]

    if not candidates:
        return None

    # ------------------------------------------------
    # 3. Version handling
    # ------------------------------------------------
    if version:
        for c in candidates:
            if c.get("version") == version:
                return str(c["id"])
        return None  # explicit version requested but not found

    # ------------------------------------------------
    # 4. No version → pick latest active
    # ------------------------------------------------
    active = [c for c in candidates if c.get("active")]
    pool = active if active else candidates

    best = max(pool, key=lambda c: Version(c["version"]))
    return str(best["id"])



from packaging.version import Version
from typing import Optional, Iterable


def find_chip(
    redis,
    query: str,
    requested_version: Optional[str] = None,
) -> Optional[Chip]:
    """
    Resolve a chip from Redis using:
    1. name / alias
    2. family
    3. version selection
    """

    q = query.lower().strip()

    # -------------------------------------------------
    # 1. Exact match via index (name or alias)
    # -------------------------------------------------
    chip_id = redis.get(CHIP_NAME_KEY.format(name=q))
    if chip_id:
        return _load_chip(redis, chip_id, requested_version)

    # -------------------------------------------------
    # 2. Family-based fallback
    # -------------------------------------------------
    chips = _load_all_chips(redis)

    family_matches = [
        c for c in chips
        if c.family and c.family.lower() == q
    ]

    if not family_matches:
        return None

    return _select_version(family_matches, requested_version)


def _load_chip(redis, chip_id: bytes, requested_version: Optional[str]) -> Chip:
    raw = redis.get(CHIP_ID_KEY.format(chip_id=int(chip_id)))
    chip = Chip.from_api(json.loads(raw))

    if not requested_version or chip.version == requested_version:
        return chip

    raise ValueError(
        f"Requested version {requested_version} not available for {chip.name}"
    )


def _load_all_chips(redis) -> list[Chip]:
    raw = redis.get(CHIP_LIST_KEY)
    if not raw:
        raise RuntimeError("Chip list not ingested")

    items = json.loads(raw)
    return [Chip.from_api(i) for i in items if i]

def _select_version(
    chips: Iterable[Chip],
    requested_version: Optional[str],
) -> Chip:
    valid = [c for c in chips if c.active]

    if not valid:
        raise ValueError("No active chips found")

    if requested_version:
        for c in valid:
            if c.version == requested_version:
                return c
        raise ValueError(f"Version {requested_version} not found")

    # Latest published
    return max(valid, key=lambda c: Version(c.version))





def _extract_family(name: str) -> Optional[str]:
    """
    Normalize chip family.

    Examples:
    - "SM8850 (KaanapalliT)" → "kaanapalli"
    - "SM8850 (KaanapalliS) r2" → "kaanapalli"
    """
    if "(" not in name:
        return None

    inside = name.split("(", 1)[1].split(")", 1)[0]
    base = inside.rstrip("TS0123456789 ").lower()

    return base or None


def _extract_alias(name: str) -> Optional[str]:
    """
    Optional alias for direct lookup.

    Examples:
    - "SM8850 (KaanapalliT)" → "kaanapallit"
    - "SM8850 (KaanapalliS) r2" → "kaanapallis"
    """
    if "(" not in name:
        return None

    inside = name.split("(", 1)[1].split(")", 1)[0]
    return inside.strip().lower() or None

class PolicyOrchestrator:
    """
    Thin decision-making layer.
    No parsing.
    No Redis schema logic.
    No XML knowledge.
    """

    def __init__(self, redis_client, ipcat_client):
        self.redis = redis_client
        self.client = ipcat_client

    # ==========================================================
    # PUBLIC ENTRY POINT
    # ==========================================================
    async def get_policy_by_mpu(
        self,
        chip_name: str,
        version: str,
        mpu_name: str,
    ) -> Dict[str, Any]:
        """
        Main API entry point.
        """

        chip_name = chip_name.lower()
        mpu_name = mpu_name.upper()

        chip_id = self._resolve_chip_id(chip_name)
        if not chip_id:
            raise ValueError(f"No chip found for chipname: {chip_name}")

        await self._ensure_ingested(chip_name, version)

        key = MPU_KEY.format(
            chip=chip_name,
            version=version,
            mpu=mpu_name,
        )

        raw = self.redis.get(key)
        if not raw:
            raise KeyError(f"MPU not found: {mpu_name}")

        return json.loads(raw)

    # ==========================================================
    # INGESTION ORCHESTRATION
    # ==========================================================
    async def _ensure_ingested(self, chip: str, version: str) -> None:
        """
        Cache-miss / TTL based ingestion trigger.
        """

        done_key = INGEST_DONE_KEY.format(chip=chip, version=version)
        lock_key = INGEST_LOCK_KEY.format(chip=chip, version=version)

        # Fast path
        if self.redis.exists(done_key):
            return

        # Someone else is ingesting
        if not self.redis.set(lock_key, 1, nx=True, ex=LOCK_TTL):
            logger.info("Ingestion already in progress: %s %s", chip, version)
            return

        try:
            logger.info("Starting ingestion: %s %s", chip, version)
            await self._ingest_chip(chip, version)
            self.redis.set(done_key, 1, ex=DONE_TTL)

        finally:
            self.redis.delete(lock_key)

    async def _ingest_chip(self, chip: str, version: str) -> None:
        """
        Fetch XML → list MPUs → ingest each MPU (threaded).
        """

        xml_text = await fetch_chip_xml(self.client, chip, version)
        mpus = extract_mpus(xml_text)

        loop = asyncio.get_running_loop()
        tasks = []

        for mpu in mpus:
            tasks.append(
                loop.run_in_executor(
                    EXECUTOR,
                    ingest_single_mpu,
                    xml_text,
                    chip,
                    version,
                    mpu,
                    self.redis,
                )
            )

        # Fail fast if any MPU fails
        for t in asyncio.as_completed(tasks):
            await t

        logger.info("Ingestion complete: %s %s", chip, version)

    # ==========================================================
    # CHIP RESOLUTION
    # ==========================================================
    def _resolve_chip_id(self, chip_name: str) -> str | None:
        """
        Resolve chip alias → chip_id via Redis.
        """
        key = CHIP_ALIAS_KEY.format(name=chip_name)
        val = self.redis.get(key)
        return val.decode() if isinstance(val, bytes) else val

    # ==========================================================
    # CHIP LIST (TTL-DRIVEN)
    # ==========================================================
    async def ensure_chip_list(self) -> None:
        """
        Refresh chip list if TTL expired.
        """

        if self.redis.exists(CHIP_LIST_KEY):
            return

        logger.info("Refreshing chip list")

        chips = await self.client.list_chips()
        payload = json.dumps(chips)

        self.redis.setex(CHIP_LIST_KEY, 36 * 3600, payload)

        for c in chips:
            for name in filter(None, [c["name"], c.get("alias")]):
                self.redis.setex(
                    CHIP_ALIAS_KEY.format(name=name.lower()),
                    36 * 3600,
                    c["id"],
                )



from concurrent.futures import ThreadPoolExecutor, as_completed
import xml.etree.ElementTree as ET

def ingest_mpus(
    xml_text: str,
    chip: str,
    version: str,
    redis_client,
    max_workers: int = 8,
) -> None:
    root = ET.fromstring(xml_text)
    mpu_elems = root.findall(".//MPU")

    with ThreadPoolExecutor(max_workers=max_workers) as executor:
        futures = [
            executor.submit(
                ingest_single_mpu,
                mpu_elem,
                chip,
                version,
                redis_client,
            )
            for mpu_elem in mpu_elems
        ]

        # Fail fast if any MPU ingestion fails
        for future in as_completed(futures):
            future.result()

def ingest_single_mpu(
    mpu_elem: ET.Element,
    chip: str,
    version: str,
    redis_client,
) -> None:
    mpu_name = mpu_elem.attrib["name"]

    hw = extract_hw_parameters(mpu_elem)

    redis_key = f"ipcat:mpu:{chip}:{version}:{mpu_name}"

    redis_client.set(
        redis_key,
        json.dumps({
            "mpu": mpu_name,
            "chip": chip,
            "version": version,
            "hw": hw,
        })
    )

from typing import Dict
import xml.etree.ElementTree as ET

def extract_hw_parameters(mpu_elem: ET.Element) -> Dict[str, Dict[str, str]]:
    """
    Extract HWDesignParameters and HWIntegrationParameters
    for a single MPU element.
    """

    result = {
        "design": {},
        "integration": {},
    }

    hw_cfg = mpu_elem.find("HWConfig")
    if hw_cfg is None:
        return result

    design = hw_cfg.find("HWDesignParameters")
    if design is not None:
        for p in design.findall("Parameter"):
            name = p.attrib.get("name")
            value = p.attrib.get("value")
            if name and value is not None:
                result["design"][name] = value

    integration = hw_cfg.find("HWIntegrationParameters")
    if integration is not None:
        for p in integration.findall("Parameter"):
            name = p.attrib.get("name")
            value = p.attrib.get("value")
            if name and value is not None:
                result["integration"][name] = value

    return result

            if r.status == 401:
                await self._fetch_token(session)
                return await self.get(session, path)
            return await r.json()

def ensure_chips(self):
    """
    Ensures chip list exists in Redis.
    Auto-heals Redis if schema version changed.
    """

    cached_schema = redis_client.get(CHIP_SCHEMA_KEY)
    if cached_schema != CHIP_SCHEMA_VERSION:
        logger.warning(
            "Chip schema drift detected (redis=%s, code=%s). Flushing chip cache.",
            cached_schema,
            CHIP_SCHEMA_VERSION,
        )
        self._flush_chip_cache()

    if redis_client.exists(CHIP_LIST_KEY):
        return

    raw = self.client.list_chips_raw()  # raw JSON
    chips = []

    for item in raw:
        try:
            chip = Chip.from_api(item)
            chips.append(chip)
        except KeyError as e:
            logger.error("Skipping invalid chip entry: %s", item)
            continue

    if not chips:
        raise RuntimeError("No valid chips returned from IPCatalog")

    # Store schema version
    redis_client.set(CHIP_SCHEMA_KEY, CHIP_SCHEMA_VERSION)

    # Store chip IDs list
    redis_client.setex(
        CHIP_LIST_KEY,
        TTL_CHIPS,
        ",".join(str(c.id) for c in chips),
    )

    # Store alias → id mapping
    for chip in chips:
        redis_client.setex(
            CHIP_ALIAS_KEY.format(name=chip.name.lower()),
            TTL_CHIPS,
            chip.id,
        )

    logger.info("Cached %d chips in Redis", len(chips))

from typing import List, Dict, Any

Chip = Dict[str, Any]

def list_chips(self) -> List[Chip]:
    data = self.request("GET", "/chip")

    if not isinstance(data, list):
        raise TypeError(f"Expected list from /chip, got {type(data)}")

    valid_chips: List[Chip] = []

    for i, item in enumerate(data):
        if not isinstance(item, dict):
            raise TypeError(f"Item {i} is not a dict: {item!r}")

        # Required fields per IPCatalog API
        required_keys = {"id", "name", "family", "type"}
        missing = required_keys - item.keys()

        if missing:
            raise KeyError(
                f"Chip item {i} missing keys {missing}. Full item={item!r}"
            )

        # Validate nested structures
        if not isinstance(item["family"], dict) or "name" not in item["family"]:
            raise KeyError(f"Chip item {i} has invalid family field: {item['family']!r}")

        if not isinstance(item["type"], dict) or "name" not in item["type"]:
            raise KeyError(f"Chip item {i} has invalid type field: {item['type']!r}")

        valid_chips.append(item)

    return valid_chips





from typing import Generator, List
import time
import math
import xml.etree.ElementTree as ET


def fetch_parsed_policy_export(
    self,
    export_id: int | str,
    export_token: str,
    *,
    page_size: int = 500,
    poll_interval: int = 5,
    timeout: int = 300,
) -> Generator[List[ET.Element], None, None]:
    """
    Polls an async xPU policy export, parses XML, and yields paginated XML elements.

    Yields:
        List[ET.Element]: A page of repeating XML elements (policies / regions).

    Raises:
        TimeoutError: If export generation exceeds timeout.
        RuntimeError: If response is invalid or empty.
    """

    url = join_url(self.base_url, f"xpu/policy/export/{export_id}/")
    start = time.time()

    # ---- Poll until XML is ready ----
    while True:
        resp = self.request(
            "GET",
            api=url,
            params={"raw": True, "token": export_token},
            response="raw",
        )

        status = resp.status_code
        ctype = resp.headers.get("content-type", "").lower()

        logger.info(f"Export poll status={status}, content-type={ctype}")

        if status == 202:
            if time.time() - start > timeout:
                raise TimeoutError("Policy export generation timed out")
            time.sleep(poll_interval)
            continue

        if status != 200:
            raise RuntimeError(f"Unexpected export status {status}")

        if "xml" not in ctype:
            # still preparing file
            time.sleep(poll_interval)
            continue

        break  # XML ready

    # ---- Parse XML ----
    try:
        root = ET.fromstring(resp.content)
    except ET.ParseError as e:
        raise RuntimeError("Failed to parse policy XML") from e

    tag, elements = finding_repeating_elements(root)

    if not elements:
        logger.info("Single XML document detected (no repeating elements)")
        yield [root]
        return

    total = len(elements)
    max_pages = math.ceil(total / page_size)

    logger.info(
        "Parsed policy XML: tag=%s total_elements=%d pages=%d",
        tag,
        total,
        max_pages,
    )

    # ---- Yield paginated results ----
    for page in range(max_pages):
        start_idx = page * page_size
        end_idx = min(start_idx + page_size, total)
        yield elements[start_idx:end_idx]







def select_xpu_policy(
    policies: List[XPUPolicy],
    requested_version: Optional[str] = None
) -> Optional[XPUPolicy]:
    valid = [
        p for p in policies
        if p["published"] and not p["is_deleted"]
    ]

    if not valid:
        return None

    if requested_version:
        for p in valid:
            if p["version"] == requested_version:
                return p
        return None

    # latest published
    return max(valid, key=lambda p: Version(p["version"]))

import io
import json
import time
import zipfile
from itertools import islice


def fetch_policy_export_paginated(
    client,
    export_id,
    export_token,
    page_size=500,
    poll_interval=5,
    timeout=300,
):
    """
    Yields paginated policy records from a large async export.

    Yields:
        list[dict]  # page_size chunk
    """

    # ---- 1. Wait until export is ready ----
    start = time.time()

    while True:
        resp = client.request(
            "GET",
            api=f"xpu/policy/export/{export_id}/",
            params={"raw": True, "token": export_token},
            response="raw",
        )

        if resp.status_code == 202:
            if time.time() - start > timeout:
                raise TimeoutError("Export generation timed out")
            time.sleep(poll_interval)
            continue

        if resp.status_code != 200:
            raise RuntimeError(
                f"Export failed: {resp.status_code} - {resp.content[:200]}"
            )

        break  # ready

    # ---- 2. Load content into memory ----
    content = resp.content
    content_type = resp.headers.get("content-type", "").lower()

    # ---- 3. Extract JSON records ----
    records = None

    # Case A: ZIP export
    if "application/zip" in content_type or content.startswith(b"PK\x03\x04"):
        with zipfile.ZipFile(io.BytesIO(content)) as z:
            for name in z.namelist():
                if name.endswith(".json"):
                    with z.open(name) as f:
                        data = json.load(f)

                        # normalize structure
                        if isinstance(data, list):
                            records = data
                        elif isinstance(data, dict):
                            records = (
                                data.get("policies")
                                or data.get("items")
                                or data.get("data")
                            )

                        if records:
                            break

        if records is None:
            raise RuntimeError("No JSON records found in export ZIP")

    # Case B: Direct JSON
    else:
        data = json.loads(content.decode())
        if isinstance(data, list):
            records = data
        elif isinstance(data, dict):
            records = (
                data.get("policies")
                or data.get("items")
                or data.get("data")
            )

    if not records:
        raise RuntimeError("No paginatable records found")

    # ---- 4. Paginate (generator) ----
    it = iter(records)

    while True:
        page = list(islice(it, page_size))
        if not page:
            break
        yield page





import io

def fetch_parsed_policy_export(client, export_id, export_token):
    """
    Downloads an async policy export into memory and returns parsed policy data.

    Returns:
        dict | list | str
            Parsed policy data (JSON) or raw text if non-JSON.
    Raises:
        RuntimeError on unexpected response or empty export.
    """

    resp = client.request(
        "GET",
        api=f"xpu/policy/export/{export_id}/",
        params={
            "raw": True,
            "token": export_token,
        },
        response="raw",
    )

    if resp.status_code != 200:
        raise RuntimeError(
            f"Export download failed: {resp.status_code} - {resp.content[:200]}"
        )

    content_type = resp.headers.get("content-type", "").lower()
    content = resp.content

    # ---- Case 1: JSON directly returned ----
    if "application/json" in content_type:
        return json.loads(content.decode("utf-8"))

    # ---- Case 2: ZIP archive ----
    if "application/zip" in content_type or content.startswith(b"PK\x03\x04"):
        zip_bytes = io.BytesIO(content)

        parsed = {}

        with zipfile.ZipFile(zip_bytes) as z:
            if not z.namelist():
                raise RuntimeError("Export ZIP is empty")

            for name in z.namelist():
                with z.open(name) as f:
                    if name.endswith(".json"):
                        parsed[name] = json.load(f)
                    else:
                        parsed[name] = f.read().decode(errors="ignore")

        return parsed

    # ---- Case 3: Fallback (text / unknown) ----
    try:
        return content.decode("utf-8")
    except UnicodeDecodeError:
        return content

def regions_by_project(project: str, mpu_name: str | None = None):
    base = """
        SELECT DISTINCT region
        FROM mpu_regions
        WHERE project = :project
    """
    params = {"project": project}

    if mpu_name:
        base += " AND mpu_name = :mpu_name"
        params["mpu_name"] = mpu_name

    base += " ORDER BY region"
    return base, params

def region_count_by_project(project: str, mpu_name: str | None = None):
    base = """
        SELECT COUNT(DISTINCT region) AS count
        FROM mpu_regions
        WHERE project = :project
    """
    params = {"project": project}

    if mpu_name:
        base += " AND mpu_name = :mpu_name"
        params["mpu_name"] = mpu_name

    return base, params
