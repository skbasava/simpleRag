def _decode_region(data: bytes | str | None) -> dict | None:
    if data is None:
        return None
    if isinstance(data, bytes):
        data = data.decode('utf-8')
    return json.loads(data)


async def _search_address_in_all_mpus(
    self, chip_id, version, address, project
) -> str:

    # Convert address once
    address_int = int(address, 16) if str(address).startswith('0x') else int(address)
    
    mpu_list = await self.engine.orchestrator.get_mpu_list(str(chip_id), version)
    redis_client = self.engine.orchestrator.redis

    async def _scan_single_mpu(mpu_name: str) -> list:
        key_pattern = f"ipcat:region:{chip_id}:{version}:{mpu_name}:*"
        keys = []
        cursor = 0
        while True:
            cursor, batch = await redis_client.scan(cursor, match=key_pattern, count=100)
            keys.extend(batch)
            if cursor == 0:
                break

        if not keys:
            return []

        # Fetch all regions in one round trip
        values = await redis_client.mget(*keys)
        
        matches = []
        for raw in values:
            region = _decode_region(raw)
            if not region:
                continue
            start = int(region.get('start', 0), 16) if str(region.get('start',0)).startswith('0x') else int(region.get('start', 0))
            end = int(region.get('end', 0), 16) if str(region.get('end',0)).startswith('0x') else int(region.get('end', 0))
            if start <= address_int <= end:
                matches.append({
                    "mpu_name": mpu_name,
                    "region_index": region.get("index"),
                    "start_address": f"0x{start:x}",
                    "end_address": f"0x{end:x}",
                    "size": region.get("size"),
                    "permissions": region.get("permissions", {}),
                    "profiles": region.get("profiles", []),
                    "address_offset": address_int - start,
                    "match": "exact"
                })
        return matches

    # ✅ Scan all MPUs concurrently
    results_per_mpu = await asyncio.gather(
        *[_scan_single_mpu(mpu) for mpu in mpu_list],
        return_exceptions=True
    )

    matches = []
    for mpu_name, result in zip(mpu_list, results_per_mpu):
        if isinstance(result, Exception):
            logger.warning(f"Failed to scan MPU {mpu_name}: {result}")
        else:
            matches.extend(result)

    result = {
        "query": f"Search for address 0x{address_int:x}",
        "project": project,
        "address": f"0x{address_int:x}",
        "chip_id": chip_id,
        "version": version,
        "mpus_scanned": len(mpu_list),
        "matches_found": len(matches),
        "matches": matches
    }

    if not matches:
        result["message"] = f"Address 0x{address_int:x} not found in any MPU region"

    return json.dumps(result, indent=2)



from dataclasses import dataclass
from typing import Optional, Any, Dict
import logging

logger = logging.getLogger(__name__)


@dataclass(frozen=True)
class ChipFamily:
    id: int
    name: str
    generation: Optional[int] = None


@dataclass(frozen=True)
class ChipType:
    id: int
    name: str


@dataclass(frozen=True)
class Chip:
    id: int
    name: str
    alias: Optional[str]
    family: Optional[ChipFamily]
    type: ChipType
    baseline_chip: Optional[str] = None
    last_revised: Optional[str] = None

    # ----------------------------
    # Factory: API → Chip
    # ----------------------------
    @staticmethod
    def from_api(item: Any) -> Optional["Chip"]:
        """
        Convert IPCatalog API JSON → Chip dataclass.

        - Idempotent (Chip in → Chip out)
        - Skips invalid / drifted records safely
        """

        # ✅ Already parsed (cache-safe)
        if isinstance(item, Chip):
            return item

        # ❌ Not JSON
        if not isinstance(item, dict):
            logger.warning("Skipping chip: expected dict, got %s", type(item))
            return None

        try:
            # ---- family (nullable in API) ----
            family = None
            fam = item.get("family")
            if isinstance(fam, dict):
                family = ChipFamily(
                    id=fam["id"],
                    name=fam["name"],
                    generation=fam.get("generation"),
                )

            # ---- type (required) ----
            type_obj = item["type"]
            chip_type = ChipType(
                id=type_obj["id"],
                name=type_obj["name"],
            )

            return Chip(
                id=item["id"],
                name=item["name"],
                alias=item.get("alias"),          # VERY IMPORTANT
                family=family,
                type=chip_type,
                baseline_chip=item.get("baseline_chip"),
                last_revised=item.get("last_revised"),
            )

        except KeyError as e:
            logger.warning(
                "Skipping chip due to schema drift: missing %s | item=%r",
                e, item
            )
            return None
